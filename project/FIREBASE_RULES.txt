rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper: Check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper: Get authenticated user ID
    function getUserId() {
      return request.auth.uid;
    }
    
    // Helper: Check if user is in a specific game room
    function isPlayerInRoom(roomId) {
      return exists(/databases/$(database)/documents/gameRooms/$(roomId)) &&
        get(/databases/$(database)/documents/gameRooms/$(roomId)).data.players[getUserId()] != null;
    }
    
    // Helper: Check if user is the room creator
    function isRoomCreator(roomId) {
      return get(/databases/$(database)/documents/gameRooms/$(roomId)).data.createdBy == getUserId();
    }
    
    // ========== PLAYERS COLLECTION ==========
    match /players/{userId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && userId == getUserId();
      allow update: if isAuthenticated() && userId == getUserId();
      allow delete: if false;
    }
    
    // ========== PLAYER STATS COLLECTION ==========
    match /playerStats/{userId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && userId == getUserId();
      allow update: if isAuthenticated() && userId == getUserId();
      allow delete: if false;
    }
    
    // ========== GAME ROOMS COLLECTION ==========
    match /gameRooms/{roomId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && 
        (resource.data.players[getUserId()] != null || resource.data.createdBy == getUserId());
      allow delete: if isAuthenticated() && 
        (resource.data.createdBy == getUserId() || resource.data.expiresAt < request.time);
    }
    
    // ========== GAME STATES COLLECTION ==========
    match /gameStates/{roomId} {
      allow read: if isAuthenticated() && isPlayerInRoom(roomId);
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && isPlayerInRoom(roomId);
      allow delete: if false;
    }
  }
}
